  66     rp.sur_agree = opt[6];
  67     rp.maxnode  = pow((double)2.0, opt[7]) -1;
  68     rp.vcost    = cost;

  70     /*
  71     ** create the "ragged array" pointers to the matrix
  72     **   x and missmat are in column major order
  73     **   y is in row major order
  74     */
  75     rp.xdata = (FLOAT **) ALLOC(nvar, sizeof(FLOAT *));
  76     for (i=0; i<nvar; i++) {
  77     rp.xdata[i] = &(xmat[i*n]);
  78     }
  79     rp.ydata = (double **) ALLOC(n, sizeof(double *));
  80     for (i=0; i<n; i++)  rp.ydata[i] = &(ymat[i*rp.num_y]);
  81 
  82     /*
  83     ** allocate some scratch
  84     */
  85     rp.tempvec = (int *)ALLOC(2*n, sizeof(int));

  88     rp.ytemp = (double **)ALLOC(n, sizeof(double *));

  91     /*
  92     ** create a matrix of sort indices, one for each continuous variable

  95     ** I don't have to sort the categoricals.

  97     rp.sorts  = (Sint**) ALLOC(nvar, sizeof(Sint *));

 105             }

 113     /*

 116     if (maxcat >0) {

 128     /*

 137     */
 138     total_wt =0;
 139     for (i=0; i<rp.n; i++) total_wt += rp.wt[i];

 159         }

 161     /* rescale the cp */
 162     for (j=0; j<rp.num_unique_cp; j++) cp[j] *= temp/old_wt;
 163     rp.alpha *= temp/old_wt;
 164     old_wt = temp;

 166     /*

 168     */
 169     xtree = (struct node *) CALLOC(1, nodesize);
 170     xtree->num_obs = k;
 171     (*rp_init)(k,rp.ytemp, maxcat, error, parms, &temp, 2, rp.wtemp);
 172     (*rp_eval)(k, rp.ytemp, xtree->response_est, &(xtree->risk),

 174     xtree->complexity = xtree->risk;
 175     partition(1, xtree, &temp);

 181     for (j=0; j<rp.n; j++) {
