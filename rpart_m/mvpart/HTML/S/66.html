  58 /*
  59 ** The anova splitting function.  Find that split point in x such that
  60 **  the sum of squares of y within the two groups is decreased as much
  61 **  as possible.  It is not necessary to actually calculate the SS, the
  62 **  improvement involves only means in the two groups.
  63 */
  64 void anova(int n,    double *y[],     FLOAT *x,     int nclass, 
     /*  [+64 src/anova.c] */
  65        int edge, double *improve, FLOAT *split, int *csplit, 
  66        double myrisk,             double *wt)
  67     {

  69     double temp = 0;
  70     double left_sum, right_sum;
  71     double left_wt, right_wt;
  72     int    left_n,  right_n;
  73     double grandmean, best;
  74     int direction = LEFT;
  75     int where = 0;
  76 
  77     /*
  78     ** The improvement of a node is SS - (SS_L + SS_R), where
  79     **   SS = sum of squares in a node = \sum w_i (x_i - \bar x)^2, where
  80     ** of course \bar x is a weighted mean \sum w_i x_i / \sum w_i
  81     ** Using the identity 
  82     **    \sum w_i(x_ - \bar x)^2 = \sum w_i (x_i-c)^2 - (\sum w_i)(c-\bar x)^2
  83     ** the improvement = w_l*(left mean - grand mean)^2 
  84     **                  +w_r*(right mean- grand mean)^2
  85     ** where w_l is the sum of weights in the left node, w_r similarly. 
  86     */
  87     right_wt =0;
  88     right_n  = n;
  89     right_sum =0;
  90     for (i=0; i<n; i++) {
  91     right_sum += *y[i] * wt[i];
  92     right_wt  += wt[i];
  93     }
  94     grandmean = right_sum/right_wt;
  95 
  96     if (nclass==0) {   /* continuous predictor */
  97         left_sum=0;   /* No data in left branch, to start */
  98     left_wt =0;   left_n =0;
  99     right_sum=0;  /*after subracting grand mean, it's zero */
 100     best    = 0;
 101     for (i=0; right_n>edge; i++) {
 102         left_wt += wt[i];  
 103         right_wt -= wt[i];
 104         left_n++;
 105         right_n--;
 106         temp = (*y[i] - grandmean) * wt[i];
 107         left_sum  +=temp;
 108         right_sum -=temp;
 109         if (x[i+1] !=x[i] &&  left_n>=edge) {
 110                     if (rp.dissim==1) temp = left_sum * left_sum / left_wt  +
 111                             right_sum * right_sum / right_wt;
 112                     else if (rp.dissim==2) temp = fabs(left_sum) + fabs(right_sum);
 113                     if (temp > best) {
 114                         best=temp;
 115                         where =i;
 116                         if (left_sum < right_sum) direction = LEFT;
 117                     else direction = RIGHT;
 118                 }
 119         }
 120     }
 121 
 122     *improve =  best/ myrisk;
 123     if (best>0) {   /* found something */
 124         csplit[0] = direction;
 125         *split = (x[where] + x[where+1]) /2;
 126         }
 127     }
 128 
 129     /* 
 130     ** Categorical predictor 
 131     */
 132     else {
 133     for (i=0; i<nclass; i++) {
 134         sums[i] =0;
 135         countn[i]=0;
 136         wts[i] =0;
 137         }
 138 
 139     /* rank the classes by their mean y value */
 140     for (i=0; i<n; i++) {
 141         j = (int)x[i] -1;
 142         countn[j]++;
 143         wts[j] += wt[i];
 144         sums[j] += (*y[i] - grandmean) * wt[i];
 145         }
 146     for (i=0; i<nclass; i++)  {
 147         if (countn[i] >0) {
 148         tsplit[i] = RIGHT;
 149         mean[i] = sums[i]/ wts[i];
 150         }
 151         else tsplit[i] = 0;
 152         }
 153     graycode_init2(nclass, countn, mean);
 154 
 155     /*
 156     ** Now find the split that we want
 157     */
 158     left_wt =0; 
 159     left_sum=0; right_sum=0;  
 160     left_n = 0; 
 161     best =0;
 162     where =0;
 163     while((j=graycode()) < nclass) {
 164         tsplit[j] = LEFT;
 165         left_n += countn[j];
 166         right_n-= countn[j];
 167         left_wt += wts[j];
 168         right_wt-= wts[j];
 169         left_sum += sums[j];
 170         right_sum-= sums[j];
 171         if (left_n>=edge  &&  right_n>=edge) {
 172         if (rp.dissim==1) temp = left_sum*left_sum/left_wt  +
 173                 right_sum*right_sum/right_wt;
 174         else if (rp.dissim==2) temp = fabs(left_sum) + fabs(right_sum);
 175         if (temp > best) {
 176             best=temp;
 177                         if ((left_sum/left_wt) > (right_sum/right_wt)) {
 178                                 for (i=0; i<nclass; i++) csplit[i] = -tsplit[i];
 179                         }
 180                         else {
 181                                 for (i=0; i<nclass; i++) csplit[i] = tsplit[i];
 182                         }
 183             }
 184         }
 185     }
 186 
 187     *improve = best/ myrisk;      /* % improvement */
 188     }
 189     }
