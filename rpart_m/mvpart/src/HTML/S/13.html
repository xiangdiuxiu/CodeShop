  21 int poissoninit(int n,         double *y[], int maxcat, char **error, 
     /*  [+21 poisson.c] */
  22         double *param, int *size,   int who,    double *wt)
  23     {
  24     int i;
  25     double event, time;
  26 
  27     /*allocate memory for scratch */
  28     if (who==1 && maxcat>0) {
  29     death = (double *)ALLOC(3*maxcat, sizeof(double));
  30     rate  = death + maxcat;
  31     wtime = rate + maxcat;
  32     order = (int *) ALLOC(3*maxcat, sizeof(int));
  33     order2= order +maxcat;
  34     countn= order2 + maxcat;
  35     }
  36 
  37     /* check data */
  38     if (who==1) {
  39     for (i=0; i<n; i++) {
  40         if (y[i][0] <=0) {
  41         *error =  "Invalid time point";
  42         return(1);
  43         }
  44         if (y[i][1] < 0) {
  45         *error = "Invalid event count";
  46         return(1);
  47         }
  48         }
  49     }
  50 
  51     /* compute the overall hazard rate */
  52     event=0;
  53     time =0;
  54     for (i=0; i<n; i++) {
  55     event += y[i][1] * wt[i];
  56     time  += y[i][0] * wt[i];
  57     }
  58 
  59     /*
  60     ** Param[0] will contain the desired CV.  If is is <=0, no shrinking
  61     **   is desired.  The CV determines alpha, and beta is set so that
  62     **   the gamma prior has the correct mean.
  63     */
  64     if (param[0] <=0) {
  65     exp_alpha =0;
  66     exp_beta  =0;
  67     }
  68     else {
  69     exp_alpha =  1/(param[0]*param[0]);
  70     exp_beta =  exp_alpha /(event/time);
  71     }

  73     ** Param[1] contains the xval rule:  1=deviance, 2=square root
  74     */
  75     which_pred = param[1];
  76     if (param[1] !=1 && param[1] !=2) {
  77     *error = "Invalid error rule";

  79     }

  85 /*
  86 **  Compute the error of prediction
  87 */
  88 double poissonpred(double *y, double *lambda) {
     /*  [+88 poisson.c] */
  89     double temp, dev;
  90 
  91     if (which_pred==1) {
  92     temp = y[1];
  93     dev  = temp - *lambda*y[0];
  94     if (temp>0)
  95         dev += temp * log(*lambda * y[0] / temp);
  96     
  97     return(-2*dev);
  98     }
  99     else {
 100     /*
 101     ** A version based on square roots, which is the
 102     **   variance stabilizing transform
 103     */
 104     temp = sqrt(y[1]) - sqrt(*lambda * y[0]); /*sqrt(obs) - sqrt(exp) */
 105     return(temp*temp);
 106     }   
 107     }

 109 /*
 110 ** Compute the predicted response rate (empirical Bayes) and the
 111 **   contribution to the deviance under that rate.
 112 ** Deviance = \sum w_i[ d_i \log(d_i/ p_i) - (d_i - p_i) ]
 113 **   where p_i = predicted # events = \lambda t_i
 114 */
 115 void poissondev(int n, double **y, double *value, double *risk, double *wt)
     /*  [+115 poisson.c] */
 116     {
 117     int i;
 118     double  death, time;
 119     double lambda, dev;
 120     double temp;
 121 
 122     death =0;
 123     time =0;
 124 
 125     /*
 126     ** first get the overall estimate of lambda
 127     */
 128     for (i=0; i<n; i++) {
 129     death += y[i][1] * wt[i];
 130     time  += y[i][0] * wt[i];
 131     }
 132     lambda = (death + exp_alpha) / (time + exp_beta);
 133 
 134     dev =0;
 135     for (i=0; i<n; i++) {
 136     temp = y[i][1];
 137     dev -= (lambda*y[i][0] - temp) * wt[i];
 138     if (temp>0)
 139         dev += (temp * log(lambda * y[i][0] / temp)) * wt[i];
 140     }
 141 
 142     value[0] = lambda;
 143     value[1] = death;
 144     *risk  = -2 * dev;
 145     }

 147 /*
 148 ** The poisson splitting function.  Find that split point in x such that
 149 **  the dev within the two groups is decreased as much
 150 **  as possible.  It is not necessary to actually calculate the devs,
 151 **  as nearly everything cancels.  The search for a split does not use the
 152 **  Bayes estimate, for speed reasons.  
 153 ** With \hat\lambda = (\sum w_i d_i) / (\sum w_i t_i)
 154 **  we have  deviance(total) - [ deviance(left son) + deviance(right son)]
 155 **              =  d_l \lambda_l +  d_r \lambda_r -  d_t \lambda_t,
 156 **   where d_l = weigthed sum of deaths for the left son, d_r = right, 
 157 **   d_t = total, and lambda_l etc are the estimated response rates
 158 */
 159 void poisson(int n,       double **y,      FLOAT *x,     int nclass, 
     /*  [+159 poisson.c] */
 160          int edge,    double *improve, FLOAT *split,
 161          int *csplit, double my_risk,  double *wt)
 162     {
 163     int i,j;
 164     int    left_n, right_n;
 165     double left_time, right_time;
 166     double left_d, right_d;
 167     double dev;      /*dev of the parent node (me) */
 168     double lambda1, lambda2;
 169     double best, temp;
 170     int direction = LEFT;
 171     int where = 0;
 172     int ncat;
 173 
 174     /*
 175     ** Get the total deaths and the total time
 176     */
 177     right_d =0;
 178     right_time =0;
 179     right_n =n ;
 180     for (i=0; i<n; i++) {
 181     right_d += y[i][1] * wt[i];
 182     right_time += y[i][0] * wt[i];
 183     }
 184 
 185     /*
 186     ** Compute the overall lambda and dev
 187     */
 188     lambda2 = right_d/right_time;
 189     if (lambda2 ==0) {
 190     *improve=0;         /*no deaths to split! */
 191     return;
 192     }
 193     dev = right_d*log(lambda2);
 194 
 195     /*
 196     ** at this point we split into 2 disjoint paths
 197     */
 198     if (nclass >0) goto categorical;
 199 
 200     left_time=0;
 201     left_d =0;
 202     where = -1;
 203     best    = dev;
 204     for (i=0; i < n-edge; i++) {
 205     left_d  += y[i][1] * wt[i];
 206     right_d -= y[i][1] * wt[i];
 207     left_time  +=y[i][0]*wt[i];
 208     right_time -=y[i][0]*wt[i];
 209 
 210     if (x[i+1] !=x[i] &&  (1+i)>=edge) {
 211         lambda1 = left_d / left_time;
 212         lambda2 = right_d/ right_time;
 213         temp = 0;
 214         if (lambda1 >0) temp += left_d * log(lambda1);
 215         if (lambda2 >0) temp += right_d* log(lambda2);
 216         if (temp > best) {
 217         best=temp;
 218         where =i;
 219         if (lambda1  < lambda2) direction = LEFT;
 220                 else    direction = RIGHT;
 221         }
 222         }
 223     }
 224 
 225     *improve =  -2*(dev - best) ;
 226     if (where >= 0 ) {   /* found something */
 227     csplit[0] = direction;
 228     *split = (x[where] + x[where+1]) /2;
 229     }
 230     return;
 231 
 232 categorical:;
 233     for (i=0; i<nclass; i++) {
 234     wtime[i] =0;
 235     death[i] =0;
 236     countn[i]=0;
 237     }
 238 
 239     for (i=0; i<n; i++) {
 240     j = x[i] -1;
 241     countn[j]++;                /* number per group */
 242     death[j]+= y[i][1] * wt[i];
 243     wtime[j] += y[i][0] * wt[i];     /*sum of time */
 244     }
 245 
 246     /*
 247     ** Rank the rates  - each is scored as the number of others that it
 248     **  is smaller than.  Ignore the categories which had no representatives.
 249     */
 250     ncat=0;   /* may be less than nclass if not all categories are present */
 251     for (i=0; i<nclass; i++) {
 252     order[i]=0;
 253     if (countn[i]>0) {
 254         ncat++;
 255         rate[i] = death[i]/ wtime[i];
 256         for (j=i-1; j>=0; j--) {
 257         if (countn[j] >0) {
 258             if (rate[i]>rate[j])  order[j]++;
 259             else              order[i]++;
 260             }
 261         }
 262         }
 263     }
 264     /*
 265     ** order2 will point to the largest, second largest, etc
 266     */
 267     for (i=0; i<nclass; i++) {
 268     if (countn[i]>0) order2[order[i]]=i;
 269     }
 270 
 271     /*
 272     ** Now find the split that we want
 273     **    starting with everyone in the right hand group
 274     */
 275     left_n =0;
 276     left_d =0;
 277     left_time=0;
 278     best =dev;
 279     where =0;
 280     for (i=0; i<ncat-1; i++){
 281     j = order2[i];
 282     left_n += countn[j];
 283     right_n-= countn[j];
 284     left_time += wtime[j];
 285     right_time-= wtime[j];
 286     left_d  += death[j];
 287     right_d -= death[j];
 288     if (left_n>=edge  &&  right_n>=edge) {
 289         lambda1 = left_d / left_time;
 290         lambda2 = right_d/ right_time;
 291         temp = 0;
 292         if (lambda1 >0) temp += left_d * log(lambda1);
 293         if (lambda2 >0) temp += right_d* log(lambda2);
 294         if (temp > best) {
 295         best=temp;
 296         where =i;
 297         if (lambda1  < lambda2) direction = LEFT;
 298                 else    direction = RIGHT;
 299         }
 300         }
 301     }
 302 
 303     *improve =  -2*(dev - best) ;
 304 
 305     /* (if improve=0, csplit will never be looked at by the calling routine)*/
 306     for (i=0; i<nclass; i++)      csplit[i]=0;
 307     for (i=0; i<=where; i++)  csplit[order2[i]] = direction ;
 308     for (   ; i<ncat; i++)    csplit[order2[i]] =  -direction;
 309 
 310     }
