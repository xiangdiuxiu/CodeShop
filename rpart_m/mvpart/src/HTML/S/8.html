  88 ** move ss = 0 to total over all y vars -- gd.
  89 */
  90     for (i=0; i<n; i++) twt += wt[i];
  91

  93       /*The iteration beneatch computes sum_i sum_j ||y_i-y_j||^alpha --ZhangYet*/
  94       for(i=0; i<rp.num_y; i++) {
  95         for (j=0; j<rp.num_y; j++){
  96           ss += pow(distMatrix[i][j], inx);
  97         } /*end for j=0*/
  98       } /*end for i=0*/
  99     }else{
 100       for (j=0; j<rp.num_y; j++) {
 101         ssj = 0;
 102         temp =0;
 103 
 104         for (i=0; i<n; i++) temp += y[i][j] * wt[i];
 105         grandmean[j] = temp/twt;
 106 
 107         for (i=0; i<n; i++) {
 108           temp = y[i][j] - grandmean[j];
 109           if (rp.dissim==1) ssj += temp * temp * wt[i];
 110           else if (rp.dissim==2) ssj += fabs(temp) * wt[i];
 111         }/*end for i*/
 112         value[j] = grandmean[j];
 113 /*
 114 ** drop out individual SS for now GD
 115 **
 116 **    value[j+rp.num_y] = ssj;
 117 **
 118 */
 119         ss += ssj;
 120       }/*end for j*/
 121     }
 122     
 123     *risk = ss;
 124     }

 127 ** The mrt splitting function.  Find that split point in x such that
 128 **  the sum of squares of y within the two groups is decreased as much
 129 **  as possible.  It is not necessary to actually calculate the SS, the

 133 void mrt(int n,    double *y[],  FLOAT *x,     int nclass,
     /*  [+133 mrt.c] */
 134        int edge, double *improve, FLOAT *split, int *csplit, double myrisk, double *wt)
 135     {
 136     int i,j,k;
 137     double sumdiffs_sq;
 138     double left_sum, right_sum;
 139     double left_wt, right_wt;
 140     int left_n, right_n;
 141     double best;
 142     double temp = 0;
 143     int direction = LEFT;
 144     int where = 0;
 145 
 146     /*
 147     ** Compute the grand mean, which will be subtracted from all of the
 148     **  data elements "on the fly".  This makes the hand calculator formula
 149     **  numerically stable.
 150     ** Also get the total n
 151     */
 152 
 153     right_n = n;
 154     right_wt = 0;
 155     for (i=0; i<n; i++) right_wt += wt[i];
 156     for (j=0; j<rp.num_y; j++) {
 157             grandmean[j] = 0; tdiffs[j] = 0;
 158             for (i=0; i<n; i++) grandmean[j] += y[i][j] * wt[i];
 159             grandmean[j] /= right_wt;
 160     } /*end of for j=0. this loop is used to calculate grandmean, which is the mean of a variable.*/
 161 
 162     if (nclass==0) {
 163     right_sum = 0; left_sum = 0;
 164     left_n = 0; left_wt = 0;
 165     best = 0;
 166 
 167     for (i=0; right_n>edge; i++) {
 168         temp = 0; sumdiffs_sq = 0; left_n++;  right_n--;
 169         left_wt += wt[i]; right_wt -= wt[i];
 170 
 171         for (j=0; j<rp.num_y; j++) {
 172             diffs[j] = (y[i][j] - grandmean[j]) * wt[i];
 173             temp += diffs[j];
 174             tdiffs[j] += diffs[j];
 175             if (rp.dissim==1) sumdiffs_sq += tdiffs[j]*tdiffs[j];
 176             else if (rp.dissim==2) sumdiffs_sq += fabs(tdiffs[j]);
 177         } /*end of j=0*/
 178 
 179        left_sum  +=temp;
 180        right_sum -=temp;
 181 
 182        if (x[i+1] != x[i] &&  left_n>=edge) {
 183        if (rp.dissim==1) temp = sumdiffs_sq/left_wt + sumdiffs_sq/right_wt;
 184        else if (rp.dissim==2) temp = 2.0*sumdiffs_sq;
 185 
 186            if (temp > best) {
 187         best = temp;
 188         where = i;
 189         if (left_sum < right_sum) direction = LEFT;
 190                   else    direction = RIGHT;
 191            }
 192 
 193        } /*end of if x[i+1] != x[i]*/
 194 
 195     }
 196 
 197     *improve = best / myrisk;
 198     if (best>0) {   /* found something */
 199         csplit[0] = direction;
 200         *split = (x[where] + x[where+1]) /2;
 201         }
 202     } /*end of if nclass == 0*/
 203     else { /*this else is for if nclass == 0.*/
 204 
 205     for (i=0; i<nclass; i++) {
 206         countn[i] = 0; countwt[i] = 0;
 207                 for (j=0; j<rp.num_y; j++) mean[i+nclass*j] = 0;
 208     } /*end for i=0 i<nclass*/
 209 
 210     for (i=0; i<n; i++) {
 211             k = x[i] -1;
 212         countn[k] ++;
 213                 countwt[k] += wt[i];
 214             for (j=0; j<rp.num_y; j++) mean[k+nclass*j] += (y[i][j] - grandmean[j]) * wt[i];
 215     }
 216 
 217         for (i=0; i<nclass; i++) {
 218           for (j=0; j<rp.num_y; j++) {
 219             if (countwt[i]>0) mean[i+nclass*j] /= countwt[i];
 220           } /*end of for j=0*/
 221         } /*end of for i=0*/
 222 
 223     for (i=0; i<nclass; i++) {
 224         if (countn[i]==0) tsplit[i] = 0;
 225         else tsplit[i] = RIGHT;
 226     }
 227 
 228     /*
 229     ** Now find the split that we want
 230     */
 231     left_n = 0;  right_n = n;
 232     left_wt = 0;
 233         right_wt = 0;
 234         for (i=0; i<n; i++) right_wt += wt[i];
 235     best = 0;
 236     /*
 237     ** Insert gray code bit here
 238     */
 239 
 240 /*  if (numclass==2) graycode_init2(nclass, countn, rate);
 241 **              else graycode_init1(nclass, countn);
 242 **
 243 **     Just use graycode_init1 here -- gd
 244 */
 245 
 246     graycode_init1(nclass, countn);
 247 
 248     while((i=graycode()) < nclass) {
 249 
 250 /* item i changes groups */
 251 
 252     if (tsplit[i]==LEFT) {
 253         tsplit[i]=RIGHT;
 254         right_n += countn[i];
 255         left_n -= countn[i];
 256         right_wt += countwt[i];
 257         left_wt -= countwt[i];
 258         for (j=0; j<rp.num_y; j++) tdiffs[j] += mean[i+nclass*j] * countwt[i];
 259         }
 260     else {
 261         tsplit[i]=LEFT;
 262         right_n -= countn[i];
 263         left_n += countn[i];
 264             right_wt -= countwt[i];
 265         left_wt += countwt[i];
 266         for (j=0; j<rp.num_y; j++) tdiffs[j] -= mean[i+nclass*j] * countwt[i];
 267     }
 268 
 269     if (left_n>=edge  &&  right_n>=edge) {
 270         left_sum = 0; right_sum = 0; sumdiffs_sq = 0;
 271         for (j=0; j<rp.num_y; j++) {
 272             left_sum += tdiffs[j];
 273             right_sum -= tdiffs[j];
 274         if (rp.dissim==1) sumdiffs_sq += tdiffs[j]*tdiffs[j];
 275         else if (rp.dissim==2) sumdiffs_sq += fabs(tdiffs[j]);
 276         }
 277         if (rp.dissim==1) temp = sumdiffs_sq/left_wt + sumdiffs_sq/right_wt;
 278         else if (rp.dissim==2) temp = 2.0*sumdiffs_sq;
 279         if (temp > best) {
 280                 best = temp;
 281                 if (left_sum > right_sum)
 282                 for (j=0; j<nclass; j++) csplit[j] = tsplit[j];
 283             else
 284                 for (j=0; j<nclass; j++) csplit[j] = -tsplit[j];
 285                 }
 286         }
 287         }
 288     }/*end of else （nclass !=　0）*/
 289     *improve = best / myrisk;      /* % improvement */
 290 
 291   }

 299 double mrtpred(double *y, double *yhat)
     /*  [+299 mrt.c] */
 300     {
 301     int j;
 302     int i; /*This is used for myDist.*/
 303     double inx = rp.alpha2;
 304     double temp;
 305     temp = 0;
 306     if (rp.dissim==1) {
 307     for (j=0; j<rp.num_y; j++) temp += (y[j] - yhat[j])*(y[j] - yhat[j]);
 308     }
 309     else if (rp.dissim==2) {
 310     for (j=0; j<rp.num_y; j++) temp += fabs((y[j] - yhat[j]));
 311     }
 312     else { /*add this for myDist*/
 313       for (j=0; j<rp.num_y; j++) {
 314         for (i=0; i<rp.num_y; i++) {
 315           temp += pow(distMatrix[i][j], inx);
 316         }
 317       }
 318     }
 319     return(temp);
 320     }
