  62 void mrtss(int n, double *y[],  double *value, double *risk, double *wt)
     /*  [+62 mrt.c] */
  63     {
  64     int i,j;
  65     double temp, twt;
  66     double ss, ssj;
  67     ss = 0;
  68     twt = 0;
  69     int k;
  70     double inx = rp.alpha2; /*the alpha2 we use to compute the impurity.--ZhangYet*/
  71     
  72 /*
  73 ** move ss = 0 to total over all y vars -- gd.
  74 */
  75     for (i=0; i<n; i++) twt += wt[i];
  76     temp = 0.0; /*ZhangYet*/
  77     if(rp.dissim==3){
  78       /*The iteration beneatch computes sum_i sum_j ||y_i-y_j||^alpha --ZhangYet*/
  79       for (i=0; i<n; i++) {
  80         for (j=0; j<n; j++) {
  81           for (k = 0; k < rp.num_y; k++) {
  82             temp += y[i][k] - y[j][k];
  83           }
  84           temp = sqrt(temp);
  85           ss += pow(temp, inx);
  86         } /*end for j=0*/
  87       } /*end for i=0*/
  88     }else{
  89       for (j=0; j<rp.num_y; j++) {
  90         ssj = 0;
  91         temp =0;
  92 
  93         for (i=0; i<n; i++) temp += y[i][j] * wt[i];
  94         grandmean[j] = temp/twt;
  95 
  96         for (i=0; i<n; i++) {
  97           temp = y[i][j] - grandmean[j];
  98           if (rp.dissim==1) ssj += temp * temp * wt[i];
  99           else if (rp.dissim==2) ssj += fabs(temp) * wt[i];
 100         }/*end for i*/
 101         value[j] = grandmean[j];
 102 /*
 103 ** drop out individual SS for now GD
 104 **
 105 **    value[j+rp.num_y] = ssj;
 106 **
 107 */
 108         ss += ssj;
 109       }/*end for j*/
 110     }
 111     
 112     *risk = ss;
 113     }

 122 void mrt(int n,    double *y[],  FLOAT *x,     int nclass,
     /*  [+122 mrt.c] */
 123        int edge, double *improve, FLOAT *split, int *csplit, double myrisk, double *wt)
 124     {
 125     int i,j,k;
 126     double sumdiffs_sq;
 127     double left_sum, right_sum;
 128     double left_wt, right_wt;
 129     int left_n, right_n;
 130     double best;
 131     double temp = 0;
 132     int direction = LEFT;
 133     int where = 0;
 134     int e_i, e_j, e_k; /*compute distance -- ZhangYet*/
 135     double inx = rp.alpha2; /*compute distance --ZhangYet*/
 136     /*
 137     ** Compute the grand mean, which will be subtracted from all of the
 138     **  data elements "on the fly".  This makes the hand calculator formula
 139     **  numerically stable.
 140     ** Also get the total n
 141     */
 142 
 143     right_n = n;
 144     right_wt = 0;
 145     for (i=0; i<n; i++) right_wt += wt[i];
 146     for (j=0; j<rp.num_y; j++) {
 147             grandmean[j] = 0; tdiffs[j] = 0;
 148             for (i=0; i<n; i++) grandmean[j] += y[i][j] * wt[i];
 149             grandmean[j] /= right_wt;
 150     } /*end of for j=0. this loop is used to calculate grandmean, which is the mean of a variable.*/
 151 
 152     if (nclass==0) {
 153     right_sum = 0; left_sum = 0;
 154     left_n = 0; left_wt = 0;
 155     best = 0;
 156 
 157     if (rp.dissim == 3){
 158       for (i=0; right_n>edge; i++){
 159         temp = 0; sumdiffs_sq = 0; left_n++; right_n--;
 160         left_wt += wt[i]; right_wt -= wt[i];
 161         if (left_n < edge) break; /*if left_n < edge, no need to split --ZhangYet*/
 162         right_sum = 0;
 163         left_sum = 0;
 164         /*start to compute left_sum and right_sum */
 165         temp = 0.0;
 166         for (e_i=0; e_i<left_n; e_i++) {
 167           for (e_j=0; e_j<left_n; e_j++) {
 168             for (e_k=0; e_k<rp.num_y; e_k++) temp += (y[e_i][e_k] - y[e_j][e_k]);
 169             temp = sqrt(temp);
 170             left_sum += pow(temp, inx);
 171           }
 172         } /*compute left_sum*/
 173         temp = 0.0;
 174         for(e_i=left_n; e_i<n; e_i++){
 175           for(e_j=left_n; e_j<n; e_j++){
 176             for(e_k=0; e_k<rp.num_y; e_k++) temp += (y[e_i][e_k] - y[e_j][e_k]);
 177             temp = sqrt(temp);
 178             left_sum += pow(temp, inx);
 179           }
 180         } /*compute right_sum*/
 181         temp = left_sum / (2*left_wt) + right_sum / (2*right_wt);
 182         if (temp > best) {
 183           best = temp;
 184           where = i;
 185           if (left_sum < right_sum )
 186             direction = LEFT;
 187           else 
 188             direction = RIGHT;
 189         }
 190       } /*end of for i=0;right_n>edge*/
 191     } else {
 192       for (i=0; right_n>edge; i++) {
 193         temp = 0; sumdiffs_sq = 0; left_n++;  right_n--;
 194         left_wt += wt[i]; right_wt -= wt[i];
 195 
 196         for (j=0; j<rp.num_y; j++) {
 197             diffs[j] = (y[i][j] - grandmean[j]) * wt[i];
 198             temp += diffs[j];
 199             tdiffs[j] += diffs[j];
 200             if (rp.dissim==1) sumdiffs_sq += tdiffs[j]*tdiffs[j];
 201             else if (rp.dissim==2) sumdiffs_sq += fabs(tdiffs[j]);
 202         } /*end of j=0*/
 203 
 204        left_sum  +=temp;
 205        right_sum -=temp;
 206 
 207        if (x[i+1] != x[i] &&  left_n>=edge) {
 208        if (rp.dissim==1) temp = sumdiffs_sq/left_wt + sumdiffs_sq/right_wt;
 209        else if (rp.dissim==2) temp = 2.0*sumdiffs_sq;
 210 
 211            if (temp > best) {
 212         best = temp;
 213         where = i;
 214         if (left_sum < right_sum) direction = LEFT;
 215                   else    direction = RIGHT;
 216            }
 217 
 218        } /*end of if x[i+1] != x[i]*/
 219 
 220     }
 221 
 222     }/*end of rp.dissim != 0 (else)*/
 223     *improve = best / myrisk;
 224     if (best>0) {   /* found something */
 225         csplit[0] = direction;
 226         *split = (x[where] + x[where+1]) /2;
 227         }
 228     } /*end of if nclass == 0*/
 229     else { /*this else is for if nclass == 0.*/
 230 
 231     for (i=0; i<nclass; i++) {
 232         countn[i] = 0; countwt[i] = 0;
 233                 for (j=0; j<rp.num_y; j++) mean[i+nclass*j] = 0;
 234     } /*end for i=0 i<nclass*/
 235 
 236     for (i=0; i<n; i++) {
 237             k = x[i] -1;
 238         countn[k] ++;
 239                 countwt[k] += wt[i];
 240             for (j=0; j<rp.num_y; j++) mean[k+nclass*j] += (y[i][j] - grandmean[j]) * wt[i];
 241     }
 242 
 243         for (i=0; i<nclass; i++) {
 244           for (j=0; j<rp.num_y; j++) {
 245             if (countwt[i]>0) mean[i+nclass*j] /= countwt[i];
 246           } /*end of for j=0*/
 247         } /*end of for i=0*/
 248 
 249     for (i=0; i<nclass; i++) {
 250         if (countn[i]==0) tsplit[i] = 0;
 251         else tsplit[i] = RIGHT;
 252     }
 253 
 254     /*
 255     ** Now find the split that we want
 256     */
 257     left_n = 0;  right_n = n;
 258     left_wt = 0;
 259         right_wt = 0;
 260         for (i=0; i<n; i++) right_wt += wt[i];
 261     best = 0;
 262     /*
 263     ** Insert gray code bit here
 264     */
 265 
 266 /*  if (numclass==2) graycode_init2(nclass, countn, rate);
 267 **              else graycode_init1(nclass, countn);
 268 **
 269 **     Just use graycode_init1 here -- gd
 270 */
 271 
 272     graycode_init1(nclass, countn);
 273 
 274     while((i=graycode()) < nclass) {
 275 
 276 /* item i changes groups */
 277 
 278     if (tsplit[i]==LEFT) {
 279         tsplit[i]=RIGHT;
 280         right_n += countn[i];
 281         left_n -= countn[i];
 282         right_wt += countwt[i];
 283         left_wt -= countwt[i];
 284         for (j=0; j<rp.num_y; j++) tdiffs[j] += mean[i+nclass*j] * countwt[i];
 285         }
 286     else {
 287         tsplit[i]=LEFT;
 288         right_n -= countn[i];
 289         left_n += countn[i];
 290             right_wt -= countwt[i];
 291         left_wt += countwt[i];
 292         for (j=0; j<rp.num_y; j++) tdiffs[j] -= mean[i+nclass*j] * countwt[i];
 293     }
 294 
 295     if (left_n>=edge  &&  right_n>=edge) {
 296         left_sum = 0; right_sum = 0; sumdiffs_sq = 0;
 297         for (j=0; j<rp.num_y; j++) {
 298             left_sum += tdiffs[j];
 299             right_sum -= tdiffs[j];
 300         if (rp.dissim==1) sumdiffs_sq += tdiffs[j]*tdiffs[j];
 301         else if (rp.dissim==2) sumdiffs_sq += fabs(tdiffs[j]);
 302         }
 303         if (rp.dissim==1) temp = sumdiffs_sq/left_wt + sumdiffs_sq/right_wt;
 304         else if (rp.dissim==2) temp = 2.0*sumdiffs_sq;
 305         if (temp > best) {
 306                 best = temp;
 307                 if (left_sum > right_sum)
 308                 for (j=0; j<nclass; j++) csplit[j] = tsplit[j];
 309             else
 310                 for (j=0; j<nclass; j++) csplit[j] = -tsplit[j];
 311                 }
 312         }
 313         }
 314     }/*end of else nclass !=0*/
 315     *improve = best / myrisk;      /* % improvement */
 316 
 317   }

 325 double mrtpred(double *y, double *yhat)
     /*  [+325 mrt.c] */
 326     {
 327     int j;
 328     int i; /*This is used for myDist.*/
 329     double inx = rp.alpha2;
 330     double temp;
 331     temp = 0;
 332     if (rp.dissim==1) {
 333         for (j=0; j<rp.num_y; j++) temp += (y[j] - yhat[j])*(y[j] - yhat[j]);
 334     }
 335     else if (rp.dissim==2) {
 336         for (j=0; j<rp.num_y; j++) temp += fabs((y[j] - yhat[j]));
 337     }
 338     else{
 339       for (j=0; j<rp.num_y; j++) temp += y[j];/*a test*/
 340     }
 341     return(temp);
 342     }
