 107 void ginidev(int n, double **y, double *value, double *risk, double *wt)
     /*  [+107 gini.c] */
 108     {
 109     int i, j, max = 0;
 110     double  temp, dev;
 111 
 112     dev =0;
 113     /*
 114     ** count up number in each class
 115     */
 116     for (i=0; i<numclass; i++)  freq[i]=0;
 117     for (i=0; i<n; i++) {
 118     j = y[i][0] -1;
 119     freq[j] += wt[i];
 120     }
 121 
 122     /*
 123     ** Now compute best class and its error
 124     */
 125     for (i=0; i<numclass; i++) {  /* assume class i */
 126     temp =0;
 127     for (j=0; j<numclass; j++) {
 128         temp += freq[j] * loss[j*numclass +i] *prior[j];
 129         }
 130     if (i==0 || temp < dev) {
 131         max =i;
 132         dev = temp;
 133         }
 134     }
 135 
 136     value[0] = max +1;    /* remember: external groups start at 1 */
 137     for (i=0; i<numclass; i++) value[i+1] = freq[i];
 138     *risk  = dev;
 139     }

 156 /*
 157 ** The gini splitting function.  Find that split point in x such that
 158 **  the rss within the two groups is decreased as much
 159 **  as possible.
 160 */
 161 void gini(int n,    double *y[],     FLOAT *x,     int numcat,
     /*  [+161 gini.c] */
 162       int edge, double *improve, FLOAT *split, int *csplit, double my_risk,
 163       double *wt)
 164     { 
 165     int i,j,k;
 166     double lwt, rwt;
 167     int  rtot, ltot;
 168     int    direction = LEFT, where = 0;
 169     double total_ss,
 170        best,
 171        temp, p;
 172     double lmean, rmean;    /* used to decide direction */
 173 
 174     for (i=0; i<numclass; i++) {
 175     left[i] =0;
 176     right[i]=0;
 177     }
 178     lwt =0;  rwt=0;
 179     rtot=0;  ltot=0;
 180     for (i=0; i<n; i++) {
 181     j = *y[i] -1;
 182     rwt += aprior[j] * wt[i];    /*altered weight = prior * case_weight */
 183     right[j] += wt[i];
 184     rtot++;
 185     }
 186     total_ss =0;
 187     for (i=0; i<numclass; i++){
 188     temp = aprior[i] * right[i]/ rwt;      /* p(class=i, given node A) */
 189     total_ss += rwt * (*impurity)(temp);    /* p(A) * I(A) */
 190     }
 191     best =total_ss; /* total weight of right * impurity of right + 0 *0 */
 192 
 193     /*
 194     ** at this point we split into 2 disjoint paths
 195     */
 196     if (numcat >0) goto categorical;
 197 
 198     for (i=0;  rtot >edge; i++) {
 199     j = *y[i] -1;
 200     rwt -= aprior[j] * wt[i];
 201     lwt += aprior[j] * wt[i];
 202     rtot--;
 203     ltot++;
 204     right[j] -= wt[i];
 205     left[j]  += wt[i];
 206 
 207     if (x[i+1] != x[i] &&  (ltot>=edge)) {
 208         temp =0;
 209             lmean =0; rmean =0;
 210         for (j=0; j<numclass; j++) {
 211         p = aprior[j]*left[j]/lwt;    /* p(j | left) */
 212         temp += lwt * (*impurity)(p);      /* p(left) * I(left) */
 213         lmean += p*j;
 214                 p =  aprior[j]*right[j]/rwt;   /* p(j | right) */
 215         temp += rwt * (*impurity)(p);      /*p(right) * I(right) */
 216         rmean += p*j;
 217             }
 218         if (temp < best) {
 219         best=temp;
 220         where =i;
 221         if (lmean < rmean) direction = LEFT;
 222             else           direction = RIGHT;
 223         }
 224         }
 225     }
 226 
 227     *improve =  total_ss - best;
 228     if (*improve > 0 ) {   /* found something */
 229     csplit[0] = direction;
 230     *split = (x[where] + x[where+1]) /2;
 231     }
 232     return;
 233 
 234 categorical:;
 235     /*
 236     ** First collapse the data into a numclass x numcat array
 237     **  ccnt[i][j] = number of class i obs, category j of the predictor
 238     */
 239     for (j=0; j<numcat; j++) {
 240     awt[j] =0;
 241     countn[j]=0;
 242     for (i=0; i<numclass; i++)
 243         ccnt[i][j] =0;
 244     }
 245     for (i=0; i<n; i++) {
 246     j = *y[i] -1;
 247     k = x[i] -1;
 248     awt[k] += aprior[j] * wt[i];
 249     countn[k]++;
 250     ccnt[j][k] += wt[i];
 251     }
 252 
 253     for (i=0; i<numcat; i++){ 
 254     if (awt[i]==0) tsplit[i] =0;
 255     else {
 256         rate[i] = ccnt[0][i] / awt[i];   /* a scratch array */
 257         tsplit[i]=RIGHT;
 258         }
 259         }
 260 
 261     if (numclass==2) graycode_init2(numcat, countn, rate);
 262                 else graycode_init1(numcat, countn);
 263 
 264     while((i=graycode()) < numcat) {
 265     /* item i changes groups */
 266     if (tsplit[i]==LEFT) {
 267         tsplit[i]=RIGHT;
 268         rwt  += awt[i];
 269         lwt -= awt[i];
 270         rtot += countn[i];
 271         ltot -= countn[i];
 272         for (j=0; j<numclass; j++) {
 273         right[j] += ccnt[j][i];
 274         left[j]  -= ccnt[j][i];
 275                 }
 276         }
 277     else {
 278         tsplit[i]=LEFT;
 279         rwt -= awt[i];
 280         lwt += awt[i];
 281         rtot -= countn[i];
 282         ltot += countn[i];
 283         for (j=0; j<numclass; j++) {
 284         right[j] -= ccnt[j][i];
 285         left[j]  += ccnt[j][i];
 286         }
 287     }
 288 
 289     if (ltot>=edge  &&  rtot>=edge) {
 290         temp =0;
 291         lmean=0; rmean =0;
 292         for (j=0; j<numclass; j++) {
 293         p = aprior[j]*left[j] /lwt;
 294         temp +=  lwt * (*impurity)(p);
 295         lmean += p*j;
 296                 p =  aprior[j]*right[j]/rwt;       /* p(j | right) */
 297         temp += rwt * (*impurity)(p);      /*p(right) * I(right) */
 298         rmean += p*j;
 299             }
 300         if (temp < best) {
 301         best=temp;
 302         if (lmean < rmean)
 303             for (j=0; j<numcat; j++) csplit[j] = tsplit[j];
 304         else
 305             for (j=0; j<numcat; j++) csplit[j] = -tsplit[j];
 306             }
 307         }
 308         }
 309     *improve = total_ss - best;
 310     }
