  31 double g_manhattan(double *x, int nr, int nc, int i1, int i2)
     /*  [+31 vgdist.c] */
  32 {
  33   double dist;
  34   int count, j;
  35   
  36   dist = 0.0;
  37   count = 0;
  38   for (j=0; j<nc; j++) {
  39     if (R_FINITE(x[i1]) && R_FINITE(x[i2])) {
  40       dist += fabs( x[i1] - x[i2] );
  41       count++;
  42     }
  43     i1 += nr;
  44     i2 += nr;
  45   }
  46   if (count == 0) dist = NA_REAL;
  47   return dist;
  48 }

  74 double g_euclidean(double *x, int nr, int nc, int i1, int i2)
     /*  [+74 vgdist.c] */
  75 {
  76   double dist, dev;
  77   int count, j;
  78 
  79   count = 0;
  80   dist = 0.0;
  81   for (j=0; j<nc; j++) {
  82     if (R_FINITE(x[i1]) && R_FINITE(x[i2])) {
  83       dev = x[i1] - x[i2];
  84       dist += dev*dev;
  85       count++;

  87     i1 += nr;
  88     i2 += nr;
  89   }
  90   if (count == 0) dist = NA_REAL;
  91   return sqrt(dist);

     /*  [+94 vgdist.c] */
  95 {
  96   double numer, denom, dist;
  97   int count, j;
  98 
  99   count = 0;
 100   dist = 0.0;
 101   for (j=0; j<nc; j++) {
 102     if (R_FINITE(x[i1]) && R_FINITE(x[i2])) {
 103       if (x[i1] != 0 || x[i2] != 0) {
 104     count++;
 105     denom = x[i1] + x[i2];
 106     if (denom > 0.0) {
 107       numer = fabs(x[i1] - x[i2]);
 108       dist += numer/denom;
 109     }
 110     else {
 111       dist += R_PosInf;
 112     }
 113     }
 114     }
 115     i1 += nr;
 116     i2 += nr;
 117   }
 118   if (count == 0) dist = NA_REAL;
 119   dist /= (double)count;
 120   return dist;

 132     if (R_FINITE(x[i1]) && R_FINITE(x[i2])) {

 145 double g_kulczynski(double *x, int nr, int nc, int i1, int i2)
     /*  [+145 vgdist.c] */
 146 {
 147   double sim, dist, t1, t2;
 148   int count, j;
 149 
 150   t1 = 0.0;
 151   t2 = 0.0;
 152   count = 0;
 153   sim = 0.0;
 154   for (j=0; j<nc; j++) {
 155     if (R_FINITE(x[i1]) && R_FINITE(x[i2])) {
 156       sim += (x[i1] < x[i2]) ? x[i1] : x[i2];
 157       t1 += x[i1];
 158       t2 += x[i2];
 159       count++;
 160     }
 161     i1 += nr;
 162     i2 += nr;
 163   }
 164   if (count == 0) dist = NA_REAL;
 165   dist = 1 - sim/t1/2 - sim/t2/2;
 166   return dist;
 167 }

 169 double g_maximum(double *x, int nr, int nc, int i1, int i2)
     /*  [+169 vgdist.c] */
 170 {
 171     double dev, dist;
 172     int count, j;
 173 
 174     count = 0;
 175     dist = -10^10;
 176     for(j = 0 ; j < nc ; j++) {
 177     if (R_FINITE(x[i1]) && R_FINITE(x[i2])) {
 178         dev = fabs(x[i1] - x[i2]);
 179         if(dev > dist)
 180         dist = dev;
 181         count++;
 182     }
 183     i1 += nr;
 184     i2 += nr;
 185     }
 186   if (count == 0) dist = NA_REAL;
 187   return dist;

 190 double g_binary(double *x, int nr, int nc, int i1, int i2)
     /*  [+190 vgdist.c] */
 191 {
 192     int total, count, dist;
 193     int j;
 194 
 195     total = 0;
 196     count = 0;
 197     dist = 0;
 198     for(j = 0 ; j < nc ; j++) {
 199     if (R_FINITE(x[i1]) && R_FINITE(x[i2])) {
 200         if(x[i1] || x[i2]){
 201         count++;
 202         if( ! (x[i1] && x[i2]) ) dist++;
 203         }
 204     }
 205     total++;
 206     i1 += nr;
 207     i2 += nr;
 208     }
 209   if (count == 0) dist = NA_REAL;
 210   return (double) dist / count;
 211 }

 217 void gdistance(double *x, int *nr, int *nc, double *d, int *diag, int *method)
     /*  [+217 vgdist.c] */
 218 {
 219   int dc, i, j, ij;
 220   switch(*method) {
 221   case MANHATTAN:
 222     distfun = g_manhattan;
 223     break;
 224   case EUCLIDEAN:
 225     distfun = g_euclidean;
 226     break;
 227   case CANBERRA:
 228     distfun = g_canberra;
 229     break;
 230   case BRAY:
 231     distfun = g_bray;
 232     break;
 233   case KULCZYNSKI:
 234     distfun = g_kulczynski;
 235     break;
 236   case GOWER:
 237     distfun = g_gower;
 238     break;
 239   case MAXIMUM:
 240     distfun = g_maximum;
 241     break;
 242   case BINARY:
 243     distfun = g_binary;
 244     break;
 245  }
 246  dc = (*diag) ? 0 : 1; 
 247  ij = 0;
 248   for (j=0; j <= *nr; j++)
 249     for (i=j+dc; i < *nr; i++) {
 250       d[ij++] = distfun(x, *nr, *nc, i, j);
 251     }
 252 }
